export function isValidLadderName(name: string): boolean {
  if (name.length < 1 || name.length > 20) {
    return false;
  }

  if (name.trim().length === 0) {
    return false;
  }

  if (name.startsWith(' ') || name.endsWith(' ')) {
    return false;
  }

  return /^[a-zA-Z0-9 _-]+$/.test(name);
}
