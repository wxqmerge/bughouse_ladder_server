export function createApiKeyAuth(validKey: string): (req: any, res: any, next: () => void) => void {
  return (req: any, res: any, next: () => void): void => {
    if (req.headers['x-api-key'] === validKey) {
      next();
    } else {
      res.status(401).json({ error: 'Unauthorized' });
    }
  };
}
