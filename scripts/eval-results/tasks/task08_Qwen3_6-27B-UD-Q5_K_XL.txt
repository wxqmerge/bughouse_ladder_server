export function calculatePlayerStats(player: { gameResults: string[] | null }): { wins: number; losses: number; draws: number; winRate: number; totalGames: number } {
  const results = player.gameResults || [];
  let wins = 0;
  let losses = 0;
  let draws = 0;
  let totalGames = 0;

  for (const r of results) {
    if (typeof r !== 'string') continue;
    const dashParts = r.split('-');
    if (dashParts.length < 2) continue;
    const scoreParts = dashParts[1].split(':');
    if (scoreParts.length < 2) continue;
    const p1 = Number(scoreParts[0]);
    const p2 = Number(scoreParts[1]);
    if (isNaN(p1) || isNaN(p2)) continue;

    totalGames++;
    if (p1 > p2) wins++;
    else if (p1 < p2) losses++;
    else draws++;
  }

  const winRate = totalGames > 0 ? Number((wins / totalGames).toFixed(4)) : 0;
  return { wins, losses, draws, winRate, totalGames };
}
