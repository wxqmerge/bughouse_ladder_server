export function calculatePlayerStats(player: { gameResults: string[] | null }): { wins: number; losses: number; draws: number; winRate: number; totalGames: number } {
  let wins = 0;
  let losses = 0;
  let draws = 0;
  let totalGames = 0;

  if (player.gameResults) {
    for (const result of player.gameResults) {
      const parts = result.split('-');
      if (parts.length < 2) continue;

      const scores = parts[1].split(':');
      if (scores.length < 2) continue;

      const p1 = parseInt(scores[0], 10);
      const p2 = parseInt(scores[1], 10);

      if (isNaN(p1) || isNaN(p2)) continue;

      totalGames++;
      if (p1 > p2) {
        wins++;
      } else if (p1 < p2) {
        losses++;
      } else {
        draws++;
      }
    }
  }

  const winRate = totalGames > 0 ? Number((wins / totalGames).toFixed(4)) : 0;

  return {
    wins,
    losses,
    draws,
    winRate,
    totalGames
  };
}
